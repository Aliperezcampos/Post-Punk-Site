CREATE DATABASE CulturaPostPunk;

USE CulturaPostPunk;

CREATE TABLE Musica (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255) NOT NULL,
    enlace VARCHAR(255) NOT NULL
);

CREATE TABLE Literatura (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255) NOT NULL,
    enlace VARCHAR(255) NOT NULL
);

CREATE TABLE Arte (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255) NOT NULL,
    enlace VARCHAR(255) NOT NULL
);

CREATE TABLE Eventos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255) NOT NULL,
    enlace VARCHAR(255) NOT NULL
);
